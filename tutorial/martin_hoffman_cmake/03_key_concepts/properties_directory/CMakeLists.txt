# I'm going to ask the properties of the CMAKE_CURRENT_SOURCE_DIR directory.
# Warning!  This will produce a lot of output that will crash cmake-gui.

cmake_minimum_required(VERSION 3.4)
project (Hello)
set (SRC Hello.cpp)
add_executable (Hello ${SRC})


# It is possible to retrieve properties via get_property or get_directory_property
# https://cmake.org/cmake/help/v3.4/manual/cmake-properties.7.html

if (CMAKE_BUILD_TYPE)
   set (CONFIG ${CMAKE_BUILD_TYPE})
elseif()
   set (CONFIG "Release")
endif ()
message ("CONFIG=${CONFIG}")


get_property (SRCDIR_ADDITIONAL_MAKE_CLEAN_FILES                 DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY ADDITIONAL_MAKE_CLEAN_FILES)
get_property (SRCDIR_CACHE_VARIABLES                             DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY CACHE_VARIABLES)
get_property (SRCDIR_CLEAN_NO_CUSTOM                             DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY CLEAN_NO_CUSTOM)
get_property (SRCDIR_CMAKE_CONFIGURE_DEPENDS                     DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY CMAKE_CONFIGURE_DEPENDS)
get_property (SRCDIR_COMPILE_DEFINITIONS                         DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY COMPILE_DEFINITIONS)
get_property (SRCDIR_COMPILE_OPTIONS                             DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY COMPILE_OPTIONS)
get_property (SRCDIR_DEFINITIONS                                 DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY DEFINITIONS)
get_property (SRCDIR_EXCLUDE_FROM_ALL                            DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY EXCLUDE_FROM_ALL)
get_property (SRCDIR_IMPLICIT_DEPENDS_INCLUDE_TRANSFORM          DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY IMPLICIT_DEPENDS_INCLUDE_TRANSFORM)
get_property (SRCDIR_INCLUDE_DIRECTORIES                         DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY INCLUDE_DIRECTORIES)
get_property (SRCDIR_INCLUDE_REGULAR_EXPRESSION                  DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY INCLUDE_REGULAR_EXPRESSION)
get_property (SRCDIR_INTERPROCEDURAL_OPTIMIZATION_${CONFIG}      DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY INTERPROCEDURAL_OPTIMIZATION_${CONFIG})
get_property (SRCDIR_INTERPROCEDURAL_OPTIMIZATION                DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY INTERPROCEDURAL_OPTIMIZATION)
get_property (SRCDIR_LINK_DIRECTORIES                            DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY LINK_DIRECTORIES)
get_property (SRCDIR_LISTFILE_STACK                              DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY LISTFILE_STACK)
get_property (SRCDIR_MACROS                                      DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY MACROS)
get_property (SRCDIR_PARENT_DIRECTORY                            DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY PARENT_DIRECTORY)
get_property (SRCDIR_RULE_LAUNCH_COMPILE                         DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY RULE_LAUNCH_COMPILE)
get_property (SRCDIR_RULE_LAUNCH_CUSTOM                          DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY RULE_LAUNCH_CUSTOM)
get_property (SRCDIR_RULE_LAUNCH_LINK                            DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY RULE_LAUNCH_LINK)
get_property (SRCDIR_TEST_INCLUDE_FILE                           DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY TEST_INCLUDE_FILE)
get_property (SRCDIR_VARIABLES                                   DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VARIABLES)
get_property (SRCDIR_VS_GLOBAL_SECTION_POST_ExtensibilityGlobals DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_GLOBAL_SECTION_POST_ExtensibilityGlobals)
get_property (SRCDIR_VS_GLOBAL_SECTION_PRE_ExtensibilityGlobals  DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_GLOBAL_SECTION_PRE_ExtensibilityGlobals)


# Let's print the variables to see what we got.
message ("Properties on directory: ${CMAKE_CURRENT_SOURCE_DIR}")

message ("ADDITIONAL_MAKE_CLEAN_FILES=${SRCDIR_ADDITIONAL_MAKE_CLEAN_FILES}")
message (" ")
message ("CACHE_VARIABLES=${SRCDIR_CACHE_VARIABLES}")
message (" ")
message ("CLEAN_NO_CUSTOM=${SRCDIR_CLEAN_NO_CUSTOM}")
message ("CMAKE_CONFIGURE_DEPENDS=${SRCDIR_CMAKE_CONFIGURE_DEPENDS}")
message ("COMPILE_DEFINITIONS=${SRCDIR_COMPILE_DEFINITIONS}")
message ("COMPILE_OPTIONS=${SRCDIR_COMPILE_OPTIONS}")
message ("DEFINITIONS=${SRCDIR_DEFINITIONS}")
message ("EXCLUDE_FROM_ALL=${SRCDIR_EXCLUDE_FROM_ALL}")
message ("IMPLICIT_DEPENDS_INCLUDE_TRANSFORM=${SRCDIR_IMPLICIT_DEPENDS_INCLUDE_TRANSFORM}")
message ("INCLUDE_DIRECTORIES=${SRCDIR_INCLUDE_DIRECTORIES}")
message ("INCLUDE_REGULAR_EXPRESSION=${SRCDIR_INCLUDE_REGULAR_EXPRESSION}")
message ("INTERPROCEDURAL_OPTIMIZATION_${CONFIG}=${SRCDIR_INTERPROCEDURAL_OPTIMIZATION_${CONFIG}}")
message ("INTERPROCEDURAL_OPTIMIZATION=${SRCDIR_INTERPROCEDURAL_OPTIMIZATION}")
message ("LINK_DIRECTORIES=${SRCDIR_LINK_DIRECTORIES}")
message ("LISTFILE_STACK=${SRCDIR_LISTFILE_STACK}")
message ("MACROS=${SRCDIR_MACROS}")
message ("PARENT_DIRECTORY=${SRCDIR_PARENT_DIRECTORY}")
message ("RULE_LAUNCH_COMPILE=${SRCDIR_RULE_LAUNCH_COMPILE}")
message ("RULE_LAUNCH_CUSTOM=${SRCDIR_RULE_LAUNCH_CUSTOM}")
message ("RULE_LAUNCH_LINK=${SRCDIR_RULE_LAUNCH_LINK}")
message ("TEST_INCLUDE_FILE=${SRCDIR_TEST_INCLUDE_FILE}")
message (" ")
message ("VARIABLES=${SRCDIR_VARIABLES}")
message (" ")
message ("VS_GLOBAL_SECTION_POST_ExtensibilityGlobals=${SRCDIR_VS_GLOBAL_SECTION_POST_ExtensibilityGlobals}")
message ("VS_GLOBAL_SECTION_PRE_ExtensibilityGlobals=${SRCDIR_VS_GLOBAL_SECTION_PRE_ExtensibilityGlobals}")
message (" ")
